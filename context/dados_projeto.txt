<sistema_detalhes> 
 - Nome: Self MD (Self Management Database). 
 - O que é: Um sistema No-Code que copia as funções do MySQL para uma tela visual, sem código. 
    O usuário pode criar databases, tabelas e colunas, adicionar dados/registros, 
    gerenciar tudo o que for criado (Crud completo), dinamicamente através de uma interface visual, 
    sem escrever SQL manualmente. 
 - Stack: React (Vite), SCSS (usando variáveis e mixins), React Bits (biblioteca UI), 
    Flaticon (biblioteca de icons), Axios para chamadas API, Node.js / Spring Boot no backend e 
    MySQL para o banco de dados. 
 - Diferencial: O sistema deve criar objetos físicos (tabelas, triggers e procedures) no banco de dados 
    baseando-se no que o usuário configura no front-end. 
</sistema_detalhes>

        @Transactional
        public void gerarTabelaCompletaFisica(TabelaModel tabela) {
            StringBuilder sql = new StringBuilder();
            sql.append("CREATE TABLE ").append(DB_USER_CREATIONS).append(".`").append(tabela.getNomeTabela()).append("` (");

            List<String> definicoes = new ArrayList<>();
            List<String> constraintsRelacionamento = new ArrayList<>();

            for (ColunaModel coluna : tabela.getColunas()) {
                StringBuilder colSql = new StringBuilder();
                String tipoSql = mapearTipoSQL(coluna);

                colSql.append("`").append(coluna.getNomeColuna()).append("` ").append(tipoSql);

                Map<TipoConfig, Object> configs = coluna.getConfig();
                if (configs != null) {
                    if (isTrue(configs, TipoConfig.not_null)) colSql.append(" NOT NULL");
                    if (isTrue(configs, TipoConfig.unique)) colSql.append(" UNIQUE");

                    if (isTrue(configs, TipoConfig.auto_increment) && Boolean.TRUE.equals(coluna.getIsPrimaryKey())) {
                    colSql.append(" AUTO_INCREMENT");
                    }
                }

                if (Boolean.TRUE.equals(coluna.getIsPrimaryKey())) {
                    colSql.append(" PRIMARY KEY");
                }

                definicoes.add(colSql.toString());

                if (coluna.isForeignKey() && coluna.getRelacionamentos() != null) {
                    for (RelacionamentoModel rel : coluna.getRelacionamentos()) {
                        String fkName = "fk_" + tabela.getNomeTabela() + "_" + coluna.getNomeColuna() + "_" + System.currentTimeMillis();

                        String fk = String.format("CONSTRAINT `%s` FOREIGN KEY (`%s`) REFERENCES %s.`%s`(`%s`)",
                            fkName,
                            coluna.getNomeColuna(),
                            DB_USER_CREATIONS,
                            rel.getTabelaDestino().getNomeTabela(),
                            rel.getColunaDestino().getNomeColuna());
                        constraintsRelacionamento.add(fk);
                    }
                }
            }

            sql.append(String.join(", ", definicoes));

            if (!constraintsRelacionamento.isEmpty()) {
                sql.append(", ").append(String.join(", ", constraintsRelacionamento));
            }

            sql.append(") ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;");

            System.out.println("SQL Gerado: " + sql.toString());

            jdbcTemplate.execute(sql.toString());
        }
